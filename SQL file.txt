
CREATE TABLE ACCOUNT_DETAILS
(ID INT PRIMARY KEY not null, 
NAME VARCHAR(30),
ACC_NUMBER DOUBLE not null,
ADHAR_ID BIGINT not null,
MOBILE_NUMBER BIGINT,
ACC_TYPE VARCHAR(30),
AVL_BALANCE DECIMAL,
ADRESS VARCHAR(50)
CITY  VARCHAR(20)
PIN_CODE  VARCHAR(30)
EMAIL_ID  VARCHAR(30)
LAST_TRAN_DATE TIMESTAMP,
CREATED_DATE TIMESTAMP,
UPDATED_DATE TIMESTAMP 
CONSTRAINT UC_ACCOUNT_DETAILS UNIQUE (ID,ACC_NUMBER,ADHAR_ID)
)



CREATE SEQUENCE ACCOUNT_DETAILS_SEQ
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999999
  NO CYCLE
  CACHE 10;
